#include <iostream>
using namespace std;
char a[1000001], b[1000001], ans[1000001];
int main()
{
	int n, i, temp, w;
	struct in{ int a, b; };
	while (scanf("%d", &n) != EOF)
	{
		in c;
		for (i = n - 1; i >= 0; i--)
		{
			scanf("%d%d", &c.a, &c.b);
			a[i] = (char)(c.a + '0');
			b[i] = (char)(c.b + '0');
		}
		for (i = 0, w = 0; i<n; i++)
		{
			temp = a[i] + b[i] - '0' - '0' + w;
			ans[n - i - 1] = temp % 10 + '0';
			w = temp / 10;
		}
		puts(ans);
	}
	return 0;
}
